  const CLIENT_KEY = 'sbaw2rio5tlosdtzmk';

  function handleFile(e,type){
    const files=Array.from(e.target.files);
    if(!files.length)return;
    markUploaded(type,files.map(f=>f.name));
  }
  function simulateUpload(type){
    if(type==='video') markUploaded('video',['28_Cross28_V2A.mp4','Alive_LIVE_11.mov']);
    else markUploaded('csv',['schedule.csv']);
  }
  function markUploaded(type,names){
    const zone=document.getElementById(type==='video'?'videoZone':'csvZone');
    const list=document.getElementById(type==='video'?'videoList':'csvList');
    zone.classList.add('uploaded');
    list.innerHTML=names.map(n=>`<div class="file-item">${n}</div>`).join('');
    if(type==='video') videosAdded=true; else csvAdded=true;
    checkReady();
  }
  function checkReady(){
    const btn=document.getElementById('postBtn');
    if(videosAdded&&csvAdded){
      btn.classList.add('ready');btn.disabled=false;
      document.getElementById('postBtnText').textContent='Publish to all platforms';
      document.getElementById('postBtnIcon').textContent='🚀';
    }
  }
  function startPosting(){
    const btn=document.getElementById('postBtn');
    btn.classList.remove('ready');btn.disabled=true;
    document.getElementById('postBtnText').textContent='Publishing…';
    document.getElementById('postBtnIcon').textContent='⏳';
    document.getElementById('postingStatus').classList.add('visible');
    [
      {id:'tiktokStatus',start:600,done:3000},
      {id:'igStatus',start:1000,done:5200},
      {id:'ytStatus',done:2400,start:400},
    ].forEach(({id,start,done})=>{
      setTimeout(()=>{
        document.getElementById(id).textContent='Uploading…';
        document.getElementById(id).className='status-badge uploading';

      setTimeout(()=>{
        document.getElementById(id).textContent='Live ✓';

          document.getElementById('successBanner').classList.add('visible');
          document.getElementById('postBtnText').textContent='All platforms live ✓';
          document.getElementById('postBtnIcon').textContent='✓';
        }
      },done);
    });
  }
